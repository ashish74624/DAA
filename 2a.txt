def f(n):
 if n<=0:
 print("error message")
 return
 fib_sequence=[0,1]
 if n==1:
 return fib_sequence[:1]
 elif n==2:
 return fib_sequence
 for i in range(2,n):
 next_num=fib_sequence[i -1]+fib_sequence[i -2] 
 fib_sequence.append(next_num)
 return fib_sequence
n= int(input("enter value of n:"))
result=f(n)
#if result is not None:
print("fibonacci sequence:", result)
