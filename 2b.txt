def binary_to_decimal(binary):
 decimal = 0
 power = 0
 while binary != 0:
 decimal += (binary % 10) * (2 ** power)
 binary //= 10
 power += 1
 return decimal
def octal_to_hexadecimal(octal):
 decimal = 0
 power = 0
 while octal != 0:
 decimal += (octal % 10) * (8 ** power)
 octal //= 10
 power += 1
 hexadecimal = ""
 hex_values = "0123456789ABCDEF"
 while decimal != 0:
 hexadecimal = hex_values[decimal % 16] + hexadecimal
 decimal //= 16
 return hexadecimal
binary_number = int(input("Enter a binary number: "))
decimal_number = binary_to_decimal(binary_number)
print("Decimal equivalent:", decimal_number)
octal_number = int(input("Enter an octal number: "))
hexadecimal_number = octal_to_hexadecimal(octal_number)
print("Hexadecimal equivalent:", hexadecimal_number)
